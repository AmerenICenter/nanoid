[docs]    def generate( self ):

        """Generates a Nanoid from stored 'size' and 'alphabet' attributes, stores in 'nanoid' attribute """

        string = ''
        for i in range(self.size):
            string += ( random.choice( self.alphabet ) )

        self.nanoid = string
